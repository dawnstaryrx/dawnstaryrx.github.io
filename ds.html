                                                                                                                                  1.2 对无序数组快速排序🧐

                                                                                                                                  使用快速排序。

                                                                                                                                  xxxxxxxxxx
                                                                                                                                  void QuickSort(int q[], int l, int r)
                                                                                                                                  {
                                                                                                                                      if (l >= r) return;
                                                                                                                                      int x = q[l + r >> 1], i = l - 1, j = r + 1;
                                                                                                                                      while (i < j)
                                                                                                                                      {
                                                                                                                                          do i++; while (q[i] < x);
                                                                                                                                          do j--; while (q[j] > x);
                                                                                                                                          if (i < j) swap(q[i], q[j]);
                                                                                                                                      }
                                                                                                                                      QuickSort(q, l, j), QuickSort(q, j + 1, r);
                                                                                                                                  }

                                                                                                                                  快排的平均时间复杂度是O(nlogn)，平均空间复杂度是O(logn)。

                                                                                                                                  快排的最坏时间复杂度是O(n^2)，最坏空间复杂度都是O(logn)。

                                                                                                                                  }

                                                                                                                                  1. B树

                                                                                                                                    1. m阶B树，每个结点有m个孩子。
                                                                                                                                    2. 每个结点最多有m-1个关键字（可以存有的键值对）。
                                                                                                                                    3. 根结点最少可以只有1个关键字。
                                                                                                                                    4. 非根结点至少有m/2个关键字。（m/2上取整-1）
                                                                                                                                    5. 每个结点中的关键字都按照从小到大的顺序排列，每个关键字的左子树中的所有关键字都小于它，而右子树中的所有关键字都大于它。
                                                                                                                                    6. 所有叶子结点都位于同一层，或者说根结点到每个叶子节点的的长度都相同。
                                                                                                                                    7. 每个结点都存有索引和数据，也就是对应的key和value。
                                                                                                                                    8. 根结点的关键字数量范围：1<=k<=m-1，非根结点的关键字数量范围：m/2<=k<=m-1。

                                                                                                                                  2. B+树

                                                                                                                                    1. B+树跟B树不同，B+树的非叶子结点不保存关键字记录的指针，只进行数据索引，这样使得B+树每个非叶子结点所能保存的关键字大大增加。
                                                                                                                                    2. B+树叶子结点保存了父节点的所有关键字记录的指针，所有数据地址必须要到叶子节点才能获取到。所以每次数据查询的次数都一样。
                                                                                                                                    3. B+树叶子节点的关键字从小到大有序排列，左边结尾数据都会保存右边节点开始数据的指针。

                                                                                                                                  5 希尔排序 shell sort

                                                                                                                                  最坏情况：O(n^(1.5))

                                                                                                                                  思想：插入排序对部分有序的序列效率高。

                                                                                                                                  1. 将元素先分组，每组内元素的下标是等差数列；
                                                                                                                                  2. 组内用插入排序；
                                                                                                                                  3. 缩小公差，重新分组。
                                                                                                                                  xxxxxxxxxx
                                                                                                                                  void shell_sort()
                                                                                                                                  {
                                                                                                                                      for (int d = n / 2; d; d /= 2)
                                                                                                                                      // for(int d = n / 3; d; d = d == 2 ? 1 : d / 3)
                                                                                                                                      {
                                                                                                                                          for (int start = 0; start < d; start ++)
                                                                                                                                          {
                                                                                                                                              for (int i = start + d; i < n; i += d) 
                                                                                                                                              {
                                                                                                                                                  int t = q[i], j = i;
                                                                                                                                                  while(j > start && q[j-d] > t)
                                                                                                                                                  {
                                                                                                                                                      q[j] = q[j - 1];
                                                                                                                                                      j -= d;
                                                                                                                                                  }
                                                                                                                                                  q[j] = t;
                                                                                                                                              }
                                                                                                                                          }
                                                                                                                                      }
                                                                                                                                  }

                                                                                                                                  时间复杂度：O(n^(1.5))

                                                                                                                                  空间复杂度：O(1)

                                                                                                                                  稳定性：不稳定
